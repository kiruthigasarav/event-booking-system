                        ->CREATE DATABASE event_booking; [database created]
                        ->USE event_booking; [use the database]
                        ->CREATE TABLE users (
                            id INT AUTO_INCREMENT PRIMARY KEY,
                            name VARCHAR(100),
                            email VARCHAR(100),
                            role VARCHAR(20)
                            );
                        ->CREATE TABLE events (
                            id INT AUTO_INCREMENT PRIMARY KEY,
                            title VARCHAR(100),
                            date DATE,
                            venue VARCHAR(100),
                            total_seats INT
                            );
                        ->CREATE TABLE bookings (
                            id INT AUTO_INCREMENT PRIMARY KEY,
                            user_id INT,
                            event_id INT,
                            seat_number VARCHAR(10),
                            FOREIGN KEY (user_id) REFERENCES users(id),
                            FOREIGN KEY (event_id) REFERENCES events(id)
                            );
                        ->SHOW TABLES; [checked the table that is created]
